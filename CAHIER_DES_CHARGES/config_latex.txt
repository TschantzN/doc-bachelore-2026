{ 
    /* --- CONFIGURATION LATEX WORKSHOP --- */

    // 1. On définit le dossier de sortie
    "latex-workshop.latex.outDir": "build",

    // 2. On utilise le visualiseur PDF intégré à VS Code (dans un onglet)
    "latex-workshop.view.pdf.viewer": "tab",

    // 3. La recette de compilation (l'ordre des outils)
    "latex-workshop.latex.recipes": [
        {
            "name": "pdflatex ➞ bibtex ➞ pdflatex × 2",
            "tools": [
                "pdflatex",
                "bibtex",
                "pdflatex",
                "pdflatex"
            ]
        }
    ],

    // 4. Configuration précise des outils pour MiKTeX
    "latex-workshop.latex.tools": [
        {
            "name": "pdflatex",
            "command": "pdflatex",
            "args": [
                "-synctex=1",
                "-interaction=nonstopmode",
                "-file-line-error",
                "-aux-directory=build",    // Spécifique MiKTeX : fichiers temporaires
                "-output-directory=.", // Fichier PDF
                "%DOC%"
            ]
        },
        {
            "name": "bibtex",
            "command": "bibtex",
            "args": [
                "build/%DOCFILE%"         // Cherche le .aux dans le dossier build
            ]
        }
    ],

    // Nettoyage automatique des fichiers inutiles après une compilation réussie
    "latex-workshop.latex.autoClean.run": "onBuilt"
}
